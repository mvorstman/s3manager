import os
import random
from pathlib import Path

root = Path.home() / "s3manager-test" / "temp-8155-uploadset"
root.mkdir(parents=True, exist_ok=True)

random.seed(42)

file_count = 2000
max_depth = 4
folder_span = 6

for i in range(file_count):
    depth = random.randint(0, max_depth)
    parts = [f"folder{random.randint(1, folder_span)}" for _ in range(depth)]
    d = root.joinpath(*parts)
    d.mkdir(parents=True, exist_ok=True)

    filename = f"file_{i:06d}.txt"
    size = random.choice([1024, 2048, 4096, 8192])

    with open(d / filename, "wb") as f:
        f.write(os.urandom(size))

print(f"Created {file_count} files under {root}")
PY
